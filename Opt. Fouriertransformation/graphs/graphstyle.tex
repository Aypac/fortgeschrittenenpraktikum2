\newcommand\addgammaplotSoneP[3][\@empty] { %
	\addplot[#1] table[#2, skip first n=2, x index=0, y expr={20*log10(sqrt((\thisrowno{1})^2+(\thisrowno{2})^2))}] {#3};

}
